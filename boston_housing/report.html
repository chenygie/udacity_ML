def performance_metric2(y_true, y_predict):
    """"""
    import numpy as np
    
    arr_true = np.array(y_true)
    y_mean = np.mean(y_true)
#     
    ssreg = 0 
#     
    ssres = 0
#     
    sstot = 0
    
    for i in y_predict:
        ssreg += (i - y_mean)**2
        
    for index,item in enumerate(y_true):
        ssres += (item - y_predict[index])**2
    for i in y_true:
        sstot += (i-y_mean)**2
    
    
    score = 1 - (ssres/sstot)

    return score
